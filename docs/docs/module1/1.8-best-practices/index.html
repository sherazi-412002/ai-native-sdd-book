Chapter 1.8: ROS 2 Best Practices 2026

Introduction​

As ROS 2 continues to evolve, establishing and following best practices becomes increasingly important for creating robust, maintainable, and scalable robotic systems. This chapter presents the essential best practices for ROS 2 development in 2026, covering everything from code organization to security and deployment strategies.

These practices reflect the maturity of the ROS 2 ecosystem and incorporate lessons learned from years of real-world deployment across diverse robotic applications.

Code Organization and Structure​

1. Modular Package Design​

Design packages with clear separation of concerns:

# Good package structure
robot_navigation/
├── package.xml
├── CMakeLists.txt
├── src/
│   ├── navigation_node.cpp
│   ├── path_planning.cpp
│   └── localization.cpp
├── include/
│   └── robot_navigation/
│       ├── navigation_node.hpp
│       ├── path_planning.hpp
│       └── localization.hpp
├── launch/
│   ├── navigation.launch.py
│   └── localization.launch.py
├── config/
│   ├── navigation.yaml
│   └── localization.yaml
├── test/
│   └── test_navigation.cpp
└── README.md

2. Consistent Naming Conventions​

Apply consistent naming across your codebase:

// Good: Consistent naming
class RobotController {
public:
    void initialize_robot();
    void execute_motion_plan();
    void update_sensor_data();

private:
    double robot_speed_;
    std::vector<double> sensor_readings_;
};

// Avoid: Inconsistent naming
class robot_controller {
public:
    void InitializeRobot();
    void ExecuteMotionPlan();
    void updateSensorData();

private:
    double robotSpeed_;
    std::vector<double> sensorReadings_;
};

// Example: Timeout handling
class TimeoutHandler {
public:
    bool wait_for_service(const std::string& service_name, int timeout_seconds = 5) {
        auto client = this->create_client<example_interfaces::srv::AddTwoInts>(service_name);

        // Wait with timeout
        if (!client->wait_for_service(std::chrono::seconds(timeout_seconds))) {
            RCLCPP_ERROR(this->get_logger(),

        }

        return true;
    }
};

2. Optimal Topic QoS Configuration​

Choose appropriate QoS settings for different use cases:

// Example: Appropriate QoS selection
class QoSExample : public rclcpp::Node {
public:
    QoSExample() : Node("qos_example") {
        // For sensor data (loss tolerant)
        auto sensor_qos = rclcpp::QoS(rclcpp::KeepLast(10))
            .best_effort()
            .durability_volatile();

        // For control commands (reliable)
        auto control_qos = rclcpp::QoS(rclcpp::KeepLast(1))
            .reliable()
            .deadline(std::chrono::milliseconds(100));

        // Create publishers with appropriate QoS
        sensor_publisher_ = this->create_publisher<sensor_msgs::msg::LaserScan>(
            "laser_scan", sensor_qos);

        control_publisher_ = this->create_publisher<geometry_msgs::msg::Twist>(
            "cmd_vel", control_qos);
    }

private:
    rclcpp::Publisher<sensor_msgs::msg::LaserScan>::SharedPtr sensor_publisher_;
    rclcpp::Publisher<geometry_msgs::msg::Twist>::SharedPtr control_publisher_;
};

1. Comprehensive Unit Testing​

Implement thorough unit testing:

// Example: Unit test for a ROS 2 node
#include <gtest/gtest.h>
#include <rclcpp/rclcpp.hpp>
#include "robot_navigation/path_planner.hpp"

class PathPlannerTest : public ::testing::Test {
protected:
    void SetUp() override {
        rclcpp::init(0, nullptr);
        node_ = rclcpp::Node::make_shared("test_node");
        planner_ = std::make_unique<PathPlanner>(node_);
    }

    void TearDown() override {
        rclcpp::shutdown();
    }

    rclcpp::Node::SharedPtr node_;
    std::unique_ptr<PathPlanner> planner_;
};

TEST_F(PathPlannerTest, TestPathPlanning) {
    // Test path planning functionality
    std::vector<geometry_msgs::msg::Point> waypoints;
    waypoints.emplace_back(0.0, 0.0, 0.0);
    waypoints.emplace_back(1.0, 1.0, 0.0);

    auto path = planner_->plan_path(waypoints);

    EXPECT_FALSE(path.empty());
    EXPECT_EQ(path.size(), 2);
}

TEST_F(PathPlannerTest, TestInvalidWaypoints) {
    // Test handling of invalid waypoints
    std::vector<geometry_msgs::msg::Point> invalid_waypoints;

    EXPECT_TRUE(path.empty());
}

2. Integration Testing​

Create integration tests for multi-node systems:

# Example: Integration test with multiple nodes
import unittest
import rclpy
from rclpy.node import Node
from std_msgs.msg import String

class IntegrationTest(unittest.TestCase):
    def setUp(self):
        rclpy.init()
        self.node = Node('integration_test')

        # Create publisher and subscriber
        self.publisher = self.node.create_publisher(String, 'test_topic', 10)
        self.subscriber = self.node.create_subscription(
            String, 'test_topic', self.callback, 10)

        self.received_messages = []

    def callback(self, msg):
        self.received_messages.append(msg.data)

    def test_message_flow(self):
        # Test message flow between nodes
        test_message = "Test message"
        self.publisher.publish(String(data=test_message))

        # Wait for message to be received
        rclpy.spin_once(self.node, timeout_sec=1.0)

        self.assertIn(test_message, self.received_messages)

    def tearDown(self):
        self.node.destroy_node()
        rclpy.shutdown()

Configuration Management​

1. Parameter Management Best Practices​

Implement robust parameter handling:

// Example: Parameter validation and management
class ConfigurableNode : public rclcpp::Node {
public:
    ConfigurableNode() : Node("configurable_node") {
        // Declare parameters with validation
        declare_parameter("max_speed", 1.0);
        declare_parameter("min_distance", 0.1);
        declare_parameter("enable_safety", true);

        // Validate parameters
        validate_parameters();

        // Set up parameter change callback
        setup_parameter_callbacks();
    }

private:
    void validate_parameters() {
        double max_speed = get_parameter("max_speed").as_double();
        double min_distance = get_parameter("min_distance").as_double();

        if (max_speed <= 0) {
            RCLCPP_ERROR(this->get_logger(), "Invalid max_speed parameter: %f", max_speed);
            throw std::invalid_argument("max_speed must be positive");
        }

        if (min_distance < 0) {
            RCLCPP_ERROR(this->get_logger(), "Invalid min_distance parameter: %f", min_distance);
            throw std::invalid_argument("min_distance must be non-negative");
        }
    }

    void setup_parameter_callbacks() {
        auto param_callback = [this](const std::vector<rclcpp::Parameter>& parameters) {
            for (const auto& param : parameters) {
                if (param.get_name() == "max_speed") {
                    RCLCPP_INFO(this->get_logger(), "max_speed changed to: %f", param.as_double());
                    // Update internal state
                }
            }
            return rclcpp::ParameterEvent::Result::SUCCESS;
        };

        add_on_set_parameters_callback(param_callback);
    }
};
